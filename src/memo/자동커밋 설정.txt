set autocommit true;

insert into member(member_id, money) values("data1",1000); // transaction 1
insert into member(member_id, money) values("data1",1000); // transaction 2

// commit
-> 자동 커밋 시, 한 줄 한 줄 단위로 하나의 트랜잭션이 생성/실행이 된다. 
반대로 set autocommit false의 경우 commit/rollback을 만날 때까지 하나의 트랜잭션이다. 
ex) 로그 기록 : 2개의 트랜잭션이 실행되기 때문에 하나의 쿼리문이 실행될 떄 각각 로그가 남는다
insert into member(member_id, money) values ('data1',10000);
갱신된 개수: 1
(1 ms)
insert into member(member_id, money) values ('data2',10000);
갱신된 개수: 1
(1 ms)

수동 커밋 시, commit/rollback을 해주지 않으면, 데이터 베이스에 설정된 타임 아웃을 넘어가면 자동으로 rolback이 실행이 된다.
